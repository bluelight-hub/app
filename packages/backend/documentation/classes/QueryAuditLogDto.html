  IsBoolean,
  IsEnum,
  IsArray,
  IsInt,
  IsDateString,
  Min,
  Max,
} from 'class-validator';

 * DTO für Audit-Log-Abfragen mit erweiterten Filtermöglichkeiten

  @ApiPropertyOptional({

  actionType?: AuditActionType;

  resource?: string;

    example: 'user_12345',

  @IsOptional()

  resourceId?: string;

  @IsString()
  userId?: string;

  @IsInt()

  @Min(0)

  @IsOptional()
  @IsString()
  sessionId?: string;

  @ApiPropertyOptional({
    description: 'Filter nach Request-ID',
    example: 'req_abc123def',
